% Definitions

\newtcolorbox{Definition}[1]
{
  enhanced
  ,colback = lightpurple
  ,colframe = darkpurple
  ,coltitle = darkpurple
  ,colbacktitle = lightpurple
  ,fonttitle = \bfseries
  ,title = {#1}
  ,attach boxed title to top right = {yshift = -5pt, xshift = -7mm}
  ,boxed title style = { boxrule = .25mm }
  ,arc = 5mm
  ,interior code app =
    {
      \node
      [
        anchor=south west
        ,line width = 0.5mm
        ,rounded corners
        ,inner sep = 5pt
        ,draw = darkpurple
        ,fill = lightpurple
        ,yshift = -5pt%3.5pt
        ,xshift = 7mm
        ,text = darkpurple
        ,font = \bfseries
      ] at (frame.north west)
      {\ Definition\ \null};
    }
}
